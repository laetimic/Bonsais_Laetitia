Valeur<-nrow(bonsais)

  {

    output_file <- paste0("Bonsai_", val,".html")

    output_path<- file.path(dir_output,output_file)
    print(output_path)

  
  
    quarto_render(
    input = "fiche_bonsai_v2.qmd",
